 11 /**
 12  * @class Associates depth with a node.
 13  */
 14 NodeDepth = DiffXmlUtils.createClass(null, {
 15   /**
 16    * Create a NodeDepth for the given node.
 17    *
 18    * @constructs
 19    * @param node The node to find the depth of
 20    */
 21   init: function (node) {
 22     /**
 23      * Node we're pointing to.
 24      */    
 25     this._node = node;
 26 
 27     /**
 28      * Field holding nodes depth.
 29      */
 30     this._depth = this._calculateDepth(this._node);
 31   },
 32   proto : /** @lends NodeDepth.prototype */ {
 33     
 34     /**
 35      * Returns the depth value.
 36      *
 37      * @return The current depth value
 38      */
 39     getDepth: function () {
 40       return this._depth;
 41     },
 42     
 43     /**
 44      * Returns the underlying node.
 45      *
 46      * @return The Node.
 47      */
 48     getNode: function () {
 49       return this._node;  
 50     },
 51     
 52     /**
 53      * Calculates the depth of a Node.
 54      * 
 55      * The root Node is at depth 0.
 56      * 
 57      * @param node The Node to calculate the depth of
 58      * @return The depth of the node
 59      */
 60     _calculateDepth: function (node) {
 61       var depth = 0;
 62       var tmpNode = node;
 63       var doc;
 64       if (node.nodeType == Node.DOCUMENT_NODE) {
 65         doc = node;
 66       } else {
 67         doc = tmpNode.ownerDocument;
 68       }
 69 
 70       while (tmpNode != doc) {
 71         depth++;
 72         tmpNode = tmpNode.parentNode;
 73       }
 74       
 75       return depth;
 76     }
 77   }  
 78 });
