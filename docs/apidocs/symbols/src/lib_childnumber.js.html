 11 /**
 12  * @class Class to hold and calculate DOM and XPath child numbers of node.
 13  */
 14 ChildNumber = DiffXmlUtils.createClass(null, {  
 15   /**
 16    * Default constructor.
 17    * 
 18    * @constructs
 19    * @param node Node to find the child numbers of
 20    */
 21   init: function (node) {
 22     if (node == null) {
 23       throw new Error("Node cannot be null");
 24     }
 25     
 26     if (node.parentNode == null) {
 27       throw new Error("Node must have parent");
 28     }
 29     
 30     // DOM child number.
 31     this._domChildNo = -1;
 32 
 33     // XPath child number.
 34     this._xPathChildNo = -1;
 35 
 36     // XPath char position.
 37     this._xPathCharPos = -1;
 38 
 39     // In-order DOM child number.
 40     this._inOrderDOMChildNo = -1;
 41 
 42     // In-order XPath child number.
 43     this._inOrderXPathChildNo = -1;
 44     
 45     // In-order XPath text position.
 46     this._inOrderXPathCharPos = -1;
 47     
 48     // The node we are doing the calcs on.
 49     this._node = node;
 50     
 51     // The siblings of the node and the node itself. 
 52     this._siblings = this._node.parentNode.childNodes;
 53   },
 54   proto : /** @lends ChildNumber.prototype */ {
 55     /**
 56      * Get the DOM child number.
 57      * 
 58      * @return DOM child number of associated node.
 59      */
 60     getDOM: function () {
 61       if (this._domChildNo == -1) {
 62         this._calculateDOMChildNumber();
 63       }
 64       
 65       return this._domChildNo;
 66     },
 67 
 68     /**
 69      * Get the XPath child number.
 70      * 
 71      * @return XPath child number of associated node.
 72      */
 73     getXPathCharPos: function() {
 74       if (this._xPathCharPos == -1) {
 75         this._calculateXPathChildNumberAndPosition();
 76       }
 77       
 78       return this._xPathCharPos;
 79     },
 80 
 81     /**
 82      * Get the XPath child number.
 83      * 
 84      * @return XPath child number of associated node.
 85      */
 86     getInOrderXPathCharPos: function () {
 87       if (this._inOrderXPathCharPos == -1) {
 88         this._calculateInOrderXPathChildNumberAndPosition();
 89       }
 90       
 91       return this._inOrderXPathCharPos;
 92     },
 93 
 94     /**
 95      * Get the XPath child number.
 96      * 
 97      * @return XPath child number of associated node.
 98      */
 99     getXPath: function() {
100       if (this._xPathChildNo == -1) {
101         this._calculateXPathChildNumberAndPosition();
102       }
103       
104       return this._xPathChildNo;
105     },
106 
107     /**
108      * Get the in-order XPath child number.
109      * 
110      * Only counts nodes marked in-order.
111      * 
112      * @return In-order XPath child number of associated node.
113      */
114     getInOrderXPath: function() {
115       if (this._inOrderXPathChildNo == -1) {
116         this._calculateInOrderXPathChildNumberAndPosition();
117       }
118       
119       return this._inOrderXPathChildNo;
120     },
121 
122     /**
123      * Get the in-order DOM child number.
124      * 
125      * Only counts nodes marked in-order.
126      * 
127      * @return In-order DOM child number of associated node.
128      */
129     getInOrderDOM: function () {
130       if (this._inOrderXPathChildNo == -1) {
131         this._calculateInOrderDOMChildNumber();
132       }
133       
134       return this._inOrderDOMChildNo;
135     },
136     
137     /**
138      * Determines whether XPath index should be incremented.
139      * 
140      * Handles differences between DOM index and XPath index
141      * 
142      * @param i The current position in siblings
143      * @return true If index should be incremented
144      */
145     _incIndex: function (i) {
146       var inc = true;
147       var curr = this._siblings.item(i);
148       // Handle non-coalescing of text nodes
149       if ((i > 0 && this._nodesAreTextNodes([curr, this._siblings.item(i - 1)])) || NodeOps.nodeIsEmptyText(curr)) {
150         inc = false;
151       }
152 
153       return inc;
154     },
155     
156     /**
157      * Determines whether the given Nodes are all text nodes or not.
158      * 
159      * @param nodes The Nodes to checks.
160      * @return true if all the given Nodes are text nodes
161      */
162     _nodesAreTextNodes: function(nodes) {
163       var areText = true;
164       for (var i = 0, l = nodes.length; i < l; i++) {
165         var n = nodes[i];
166         if ((n == null) || (n.nodeType != Node.TEXT_NODE)) {
167           areText = false;
168           break;
169         }
170       }
171         
172       return areText;
173     },
174 
175     /**
176      * Calculates the DOM index of the node.
177      */
178     _calculateDOMChildNumber: function () {
179       var cn;
180 
181       for (cn = 0; cn < this._siblings.length; cn++) {
182         if (NodeOps.checkIfSameNode(this._siblings.item(cn), this._node)) {
183           break;
184         }
185       }
186         
187       this._domChildNo = cn;
188     },
189 
190     /**
191      * Calculates the "in order" DOM child number of the node.
192      */
193     _calculateInOrderDOMChildNumber: function () {
194       this._inOrderDOMChildNo = 0;
195       for (var i = 0; i < this._siblings.length; i++) {
196         if (NodeOps.checkIfSameNode(this._siblings.item(i), this._node)) {
197           break;
198         }
199         
200         if (NodeOps.isInOrder(this._siblings.item(i))) {
201           this._inOrderDOMChildNo++;
202         }
203       }
204     },
205 
206     /**
207      * Sets the XPath child number and text position.
208      */
209     _calculateXPathChildNumberAndPosition: function () {
210       var domIndex = this._calculateXPathChildNumber();
211       this._calculateXPathTextPosition(domIndex);   
212     },
213 
214     /**
215      * Sets the XPath child number and text position.
216      */
217     _calculateInOrderXPathChildNumberAndPosition: function () {
218       var domIndex = this._calculateInOrderXPathChildNumber();
219       this._calculateInOrderXPathTextPosition(domIndex);   
220     },
221     
222     /**
223      * Calculate the character position of the node.
224      * 
225      * @param domIndex The DOM index of the node in its siblings.
226      */
227     _calculateXPathTextPosition: function(domIndex) {
228       this._xPathCharPos = 1;
229       for (var i = (domIndex - 1); i >= 0; i--) {
230         if (this._siblings.item(i).nodeType == Node.TEXT_NODE) {
231           this._xPathCharPos = this._xPathCharPos + this._siblings.item(i).length;
232         } else {
233           break;
234         }
235       }
236     },
237 
238     /**
239      * Set the XPath child number of the node.
240      * 
241      * @return The DOM index of the node in its siblings
242      */
243     _calculateXPathChildNumber: function () {
244       var childNo = 1;
245       var domIndex;
246       for (domIndex = 0; domIndex < this._siblings.length; domIndex++) {
247         if (NodeOps.checkIfSameNode(this._siblings.item(domIndex), this._node)) {
248           if (!this._incIndex(domIndex)) {
249             childNo--;
250           }
251           break;
252         }
253         if (this._incIndex(domIndex)) {
254           childNo++;
255         }
256       }
257       
258       this._xPathChildNo = childNo;
259         
260       return domIndex;
261     },
262 
263     /**
264      * Set the in-order XPath child number of the node.
265      * 
266      * @return The DOM index of the node in its siblings
267      */
268     _calculateInOrderXPathChildNumber: function () {
269 
270       var childNo = 0;
271       var domIndex;
272       var lastInOrderNode = null;
273       var currNode = null;
274       
275       for (domIndex = 0; domIndex < this._siblings.length; domIndex++) {
276         currNode = this._siblings.item(domIndex);
277         if (NodeOps.isInOrder(currNode) && !(this._nodesAreTextNodes([currNode, lastInOrderNode]) || NodeOps.nodeIsEmptyText(currNode))) {
278           childNo++;
279         } 
280         
281         if (NodeOps.checkIfSameNode(currNode, this._node)) {
282           break;
283         }
284         
285         if (NodeOps.isInOrder(currNode)) {
286           lastInOrderNode = currNode;
287         }
288       }
289  
290       //Add 1 if the given node wasn't in order
291       if (currNode != null && !NodeOps.isInOrder(currNode)) {
292         childNo++;
293       }
294  
295       this._inOrderXPathChildNo = childNo;
296       return domIndex;
297     },
298 
299     /**
300      * Calculate the character position of the node.
301      * 
302      * @param domIndex The DOM index of the node in its siblings.
303      */
304     _calculateInOrderXPathTextPosition: function (domIndex) {
305       this._inOrderXPathCharPos = 1;
306       
307       for (var i = (domIndex - 1); i >= 0; i--) {
308         if (this._siblings.item(i).nodeType == Node.TEXT_NODE) {
309           if (NodeOps.isInOrder(this._siblings.item(i))) { 
310             this._inOrderXPathCharPos = this._inOrderXPathCharPos + this._siblings.item(i).length;
311           }
312         } else if (NodeOps.isInOrder(this._siblings.item(i))) {
313           break;
314         }
315       }
316     }
317   }  
318 });
