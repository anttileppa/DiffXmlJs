 11 /**
 12  * @class Class to hold pairs of nodes.
 13  */
 14 NodeSequence = /** @lends NodeSequence */ {
 15   
 16   /**
 17    * Gets the nodes in set1 which have matches in set2.
 18    *
 19    * @param set1      the first set of nodes
 20    * @param set2      the set of nodes to match against
 21    * @param matchings the set of matching nodes
 22    *
 23    * @return the nodes in set1 which have matches in set2
 24    */
 25   getSequence: function (set1, set2, matchings) {
 26     var seq = null;
 27     if (set1 != null && set2 != null) {
 28       var resultSet = new Array();
 29       var set2list = new Array();
 30       for (var i = 0, l = set2.length; i < l; i++) {
 31         set2list.push(set2.item(i));
 32       }
 33       
 34       for (var i = 0, l = set1.length; i < l; i++) {
 35         if (set2list.indexOf(matchings.getPartner(set1.item(i))) != -1) {
 36           resultSet.push(set1.item(i));
 37         }   
 38       }
 39       
 40       seq = resultSet;
 41     }
 42     
 43     return seq; 
 44   },
 45 
 46   /**
 47    * Gets the Longest Common Subsequence for the given Node arrays.
 48    * 
 49    * "Matched" Nodes are considered equal.
 50    * The returned nodes are from s1.
 51    * 
 52    * TODO: Check for better algorithms
 53    * 
 54    * @param s1 First Node sequence 
 55    * @param s2 Second Node sequence
 56    * @param matchings Set of matching Nodes
 57    * @return A list of Nodes representing the Longest Common Subsequence 
 58    */
 59   getLCS: function (s1, s2, matchings) {
 60     var num = new Array();
 61     for (var i = 0, l = s1.length + 1; i < l; i++) {
 62       var subArray = new Array();
 63       for (var j = 0, jl = s2.length + 1; j < jl; j++) {
 64         subArray.push(0);
 65       }
 66       
 67       num.push(subArray);
 68     }
 69     
 70     for (var i = 1; i <= s1.length; i++) {
 71       for (var j = 1; j <= s2.length; j++) {
 72         var n1 = matchings.getPartner(s1[i - 1]);
 73         var n2 = s2[j - 1];
 74         if (NodeOps.checkIfSameNode(n1, n2)) {
 75           num[i][j] = 1 + num[i - 1][j - 1];
 76         } else {
 77           num[i][j] = Math.max(num[i - 1][j], num[i][j - 1]);
 78         }
 79       }
 80     }
 81     
 82     //Length of LCS is num[s1.length][s2.length]);
 83 
 84     var s1position = s1.length; 
 85     var s2position = s2.length;
 86     
 87     var result = new Array();
 88 
 89     while (s1position != 0 && s2position != 0) {
 90       if (NodeOps.checkIfSameNode(matchings.getPartner(s1[s1position - 1]), s2[s2position - 1])) {
 91         result.unshift(s1[s1position - 1]);
 92         s1position--;
 93         s2position--;
 94       } else if (num[s1position][s2position - 1] >= num[s1position - 1][s2position]) {
 95         s2position--;
 96       } else {
 97         s1position--;
 98       }
 99     }
100 
101     return result;
102   }   
103 };
