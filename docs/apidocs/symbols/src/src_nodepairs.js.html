 11 /**
 12  * @class Class to hold pairs of nodes.
 13  */
 14 NodePairs = DiffXmlUtils.createClass(null, {
 15   /**
 16    * Constructor
 17    * @constructs
 18    */
 19   init: function () {
 20     this._pairs = new Object();
 21     this._pairCount = 0;
 22     this._hashCounter = new Date().getTime();
 23   },
 24   proto : /** @lends NodePairs.prototype */ {
 25     /**
 26      * Adds a pair of nodes to the set. Sets UserData as matched.
 27      * 
 28      * @param x first node
 29      * @param y partner of first node
 30      */
 31     add: function (x, y) {
 32       var xHash = ++this._hashCounter; 
 33       var yHash = ++this._hashCounter; 
 34     
 35       this._pairs[xHash] = y;
 36       this._pairs[yHash] = x;
 37       this._pairCount += 2;
 38       
 39       NodeOps.setUserData(x, "hash", xHash);
 40       NodeOps.setUserData(y, "hash", yHash);
 41 
 42       this._setMatched(x, y);
 43     },
 44 
 45     /**
 46      * Mark the node as being "matched".
 47      *
 48      * @param n the node to mark as "matched"
 49      */
 50     _setMatchedNode: function (n) {
 51       NodeOps.setUserData(n, "matched", true);
 52     },
 53 
 54     /**
 55      * Check if node is marked "matched".
 56      *
 57      * Made static so that I can use a instance method later if it is faster or
 58      * better.
 59      * 
 60      * @param n node to check
 61      * @return true if marked "matched", false otherwise
 62      */
 63     isMatched: function (n) {
 64       var data = NodeOps.getUserData(n, "matched");
 65       return data == null ? false : data;
 66     },
 67     
 68     /**
 69      * Mark a pair of nodes as matched.
 70      *
 71      * @param nodeA  The unmatched partner of nodeB
 72      * @param nodeB  The unmatched partner of nodeA
 73      */
 74     _setMatched: function (nodeA, nodeB) {
 75       this._setMatchedNode(nodeA);
 76       this._setMatchedNode(nodeB);
 77     },
 78     
 79     /**
 80      * Returns the partner of a given node. Returns null if the node does not
 81      * exist.
 82      * 
 83      * @param n the node to find the partner of.
 84      * @return the partner of n.
 85      */
 86     getPartner: function (n) {
 87       if (n == null) {
 88         return null;
 89       } else {
 90         var hash = NodeOps.getUserData(n, "hash");
 91         return hash ? this._pairs[hash]||null : null;
 92       }
 93     },
 94 
 95     /**
 96      * Get the number of nodes stored. 
 97      * 
 98      * Note that this includes both nodes and partners.
 99      * 
100      * @return The number of nodes stored.
101      */
102     size: function () {
103       return this._pairCount;
104     },
105     
106     /**
107      * Remove a node and it's partner from the list of matchings.
108      * 
109      * @param n The Node to remove
110      */
111     remove: function (n) {
112       var nHash = NodeOps.getUserData(n, "hash");
113       var nMatch = this._pairs[nHash];
114       NodeOps.setUserData(nMatch, "matched", null);
115       NodeOps.setUserData(n, "matched", null);
116       
117       delete this._pairs[NodeOps.getUserData(nMatch, "hash")];
118       delete this._pairs[nHash];
119       
120       this._pairCount -= 2;
121     }
122     
123   }
124 });
