 11 /**
 12  * @class
 13  */
 14 InternalPatch = DiffXmlUtils.createClass(null, {
 15 
 16   init : function() {
 17 
 18   },
 19   proto : /** @lends InternalPatch.prototype */
 20   {

 22     /**
 23      * Apply patch to XML document.
 24      * 
 25      * @param doc
 26      *          the XML document to be patched
 27      * @param patch
 28      *          the patch
 29      * @throws PatchFormatException
 30      *           if there is an error parsing the patch
 31      */
 32     apply : function(doc, patch) {
 33 
 34       for ( var i = 0, l = patch.getChanges().length; i < l; i++) {
 35         // Normalize is essential for deletes to work
 36         doc.normalize();
 37         
 38         var operation = patch.getChanges()[i];
 39         switch (operation.type) {
 40           case 'insert':
 41             this._doInsert(doc, operation);
 42           break;
 43           case 'delete':
 44             this._doDelete(doc, operation);
 45           break;
 46           case 'move':
 47             this._doMove(doc, operation);
 48           break;
 49           case 'update':
 50             this._doUpdate(doc, operation);
 51           break;
 52         }
 53       }
 54 
 55     },
 56 
 57     /**
 58      * Apply insert operation to document.
 59      * 
 60      * @param doc
 61      *          the document to be patched
 62      * @param operation
 63      *          the insert operation node
 64      */
 65     _doInsert : function(doc, operation) {
 66       var charpos = parseInt(operation.charpos || 1);
 67       var parentNode = this._findNodeByXPath(doc, operation.parent);
 68       var siblings = parentNode.childNodes;
 69       var nodeType = operation.nodeType;
 70       var value = operation.value;
 71       var childNo = operation.childNo === undefined ? 1 : parseInt(operation.childNo);
 72       var domChildNo = this._getDOMChildNo(nodeType, siblings, childNo);
 73       var ins = null;
 74 
 75       switch (nodeType) {
 76         case Node.TEXT_NODE:
 77           ins = doc.createTextNode(value);
 78           this._insertNode(siblings, parentNode, domChildNo, charpos, ins, doc);
 79         break;
 80         case Node.CDATA_SECTION_NODE:
 81           ins = doc.createCDATASection(value);
 82           this._insertNode(siblings, parentNode, domChildNo, charpos, ins, doc);
 83         break;
 84         case Node.ELEMENT_NODE:
 85           ins = doc.createElement(operation.nodeName);
 86           this._insertNode(siblings, parentNode, domChildNo, charpos, ins, doc);
 87         break;
 88         case Node.COMMENT_NODE:
 89           ins = doc.createComment(value);
 90           this._insertNode(siblings, parentNode, domChildNo, charpos, ins, doc);
 91         break;
 92         case Node.ATTRIBUTE_NODE:
 93           if (parentNode.nodeType != Node.ELEMENT_NODE) {
 94             throw new Error("Parent not an element");
 95           }
 96 
 97           parentNode.setAttribute(operation.nodeName, value);
 98         break;
 99         default:
100           throw new Error("Unknown NodeType " + nodeType);
101       }
102     },
103 
104     /**
105      * Apply delete operation.
106      * 
107      * @param doc
108      *          document to be patched
109      * @param operation
110      *          delete operation
111      */
112     _doDelete : function(doc, operation) {
113       var deleteNode = this._findNodeByXPath(doc, operation.node);
114       if (deleteNode.nodeType == Node.ATTRIBUTE_NODE) {
115         deleteNode.ownerElement.removeAttributeNode(deleteNode);
116       } else if (deleteNode.nodeType == Node.TEXT_NODE) {
117         var charpos = operation.charpos === undefined ? 1 : operation.charpos;
118         var length = operation.length;
119         if (length === undefined) {
120           this._deleteText2(deleteNode, charpos, doc);
121         } else {
122           try {
123             this._deleteText(deleteNode, charpos, length, doc);
124           } catch (e) {
125             this._deleteText2(deleteNode, charpos, doc);
126           }
127         }
128       } else {
129         deleteNode.parentNode.removeChild(deleteNode);
130       }
131     },
132 
133     /**
134      * Perform update operation.
135      * 
136      * @param doc
137      *          The document being patched
138      * @param operation
139      *          The update operation
140      */
141     _doUpdate : function(doc, operation) {
142       var node = this._findNodeByXPath(doc, operation.node);
143 
144       if (node.nodeType == Node.ELEMENT_NODE) {
145         var newNode = doc.createElement(operation.nodeName);
146         for ( var i = 0, l = node.attributes.length; i < l; i++) {
147           newNode.attributes[node.attributes[i].name] = node.attributes[i].value;
148         }
149 
150         // Move all the children over
151         while (node.hasChildNodes()) {
152           newNode.appendChild(updateNode.firstChild);
153         }
154 
155         node.parentNode.replaceChild(newNode, node);
156       } else {
157         node.nodeValue = operation.nodeValue;
158       }
159     },
160 
161     /**
162      * Apply move operation.
163      * 
164      * @param doc
165      *          document to be patched
166      * @param operation
167      *          move operation
168      */
169     _doMove : function(doc, operation) {
170       var node = this._findNodeByXPath(doc, operation.node);
171       var oldCharPos = operation.ocharpos;
172       var newCharPos = operation.ncharpos;
173 
174       if (node.nodeType == Node.TEXT_NODE) {
175         var text = "";
176         try {
177           var length = operation.length;
178           if (length) {
179             text = this._deleteText(node, oldCharPos, length, doc);
180           } else {
181             text = this._deleteText2(node, oldCharPos, doc);
182           }
183         } catch (e) {
184           text = this._deleteText2(node, oldCharPos, doc);
185         }
186 
187         node = doc.createTextNode(text);
188       }
189 
190       if (node.nodeType != Node.TEXT_NODE) {
191         node = node.parentNode.removeChild(node);
192       }
193 
194       // Find position to move to
195       // Get parent
196       var parent = this._findNodeByXPath(doc, operation.parent);
197       var newSiblings = parent.childNodes;
198       var domcn = this._getDOMChildNo(node.nodeType, newSiblings, operation.childNo);
199 
200       // Perform insert
201       this._insertNode(newSiblings, parent, domcn, newCharPos, node, doc);
202     },
203 
204     _findNodeByXPath : function(document, xpath) {
205       var result = document.evaluate(xpath, document, null, XPathResult.ANY_TYPE, null);
206       return result.iterateNext();
207     },
208 
209     /**
210      * Get the DOM Child number
211      * 
212      * @param nodeType
213      *          the nodeType to be inserted
214      * @param siblings
215      *          the siblings of the node
216      * @param xPathChildNo
217      *          operation childNo
218      * @return the DOM Child number of the node
219      */
220     _getDOMChildNo : function(nodeType, siblings, xpathcn) {
221       var domcn = 0;
222 
223       //Convert xpath childno to DOM childno
224       if (nodeType != Node.ATTRIBUTE_NODE) {
225           domcn = this._getDOMChildNoFromXPath(siblings, xpathcn);
226       }
227 
228       return domcn;
229     },
230     
231     /**
232      * Get the DOM Child Number equivalent of the XPath childnumber.
233      *
234      * @param siblings the NodeList we are interested in
235      * @param xpathcn  the XPath child number
236      * @return the equivalent DOM child number
237      */
238     _getDOMChildNoFromXPath: function (siblings, xpathcn) {
239       var domIndex = 0;
240       var xPathIndex = 1;
241       while ((xPathIndex < xpathcn) && (domIndex < siblings.length)) {
242         if (!((this._prevNodeIsATextNode(siblings, domIndex)) && (siblings.item(domIndex).nodeType == Node.TEXT_NODE))) {
243           xPathIndex++;
244         }
245         domIndex++;
246       }
247       //Handle appending nodes
248       if (xPathIndex < xpathcn) {
249         domIndex++;
250         xPathIndex++;
251       }
252       
253       return domIndex;
254     },
255 
256     /**
257      * Tests if previous node is a text node.
258      * 
259      * @param siblings
260      *          siblings of current node
261      * @param index
262      *          index of current node
263      * @return true if previous node is a text node, false otherwise
264      */
265     _prevNodeIsATextNode : function(siblings, index) {
266       return index > 0 && siblings.item(index - 1).nodeType == Node.TEXT_NODE;
267     },
268 
269     /**
270      * Inserts a node at the given character position.
271      * 
272      * @param charpos
273      *          the character position to insert at
274      * @param siblings
275      *          the NodeList to insert the node into
276      * @param domcn
277      *          the child number to insert the node as
278      * @param ins
279      *          the node to insert
280      * @param parent
281      *          the node to become the parent of the inserted node
282      */
283     _insertAtCharPos : function(charpos, siblings, domcn, ins, parent, doc) {
284       // we know text node at domcn -1
285       var cp = charpos;
286       var textNodeIndex = domcn - 1;
287       var append = false;
288 
289       while (this._prevNodeIsATextNode(siblings, textNodeIndex)) {
290         textNodeIndex--;
291       }
292 
293       while ((siblings.item(textNodeIndex).nodeType == Node.TEXT_NODE) && (cp > siblings.item(textNodeIndex).length)) {
294         cp = cp - siblings.item(textNodeIndex).length;
295         textNodeIndex++;
296 
297         if (textNodeIndex == siblings.length) {
298           if (cp > 1) {
299             throw new Error("charpos past end of text");
300           }
301           append = true;
302           parent.appendChild(ins);
303           break;
304         }
305       }
306       ;
307 
308       var sibNode = siblings.item(textNodeIndex);
309 
310       if (!append) {
311         if (cp == 1) {
312           parent.insertBefore(ins, sibNode);
313         } else if (cp > sibNode.length) {
314           var nextSib = sibNode.nextSibling;
315           if (nextSib != null) {
316             parent.insertBefore(ins, nextSib);
317           } else {
318             parent.appendChild(ins);
319           }
320         } else {
321           var text = sibNode.nodeValue;
322           var nextSib = sibNode.nextSibling;
323           parent.removeChild(sibNode);
324           var text1 = doc.createTextNode(text.substring(0, cp - 1));
325           var text2 = doc.createTextNode(text.substring(cp - 1));
326           if (nextSib != null) {
327             parent.insertBefore(text1, nextSib);
328             parent.insertBefore(ins, nextSib);
329             parent.insertBefore(text2, nextSib);
330           } else {
331             parent.appendChild(text1);
332             parent.appendChild(ins);
333             parent.appendChild(text2);
334           }
335         }
336       }
337     },
338 
339     /**
340      * Insert a node under parent node at given position.
341      * 
342      * @param siblings
343      *          the NodeList to insert the node into
344      * @param parent
345      *          the parent to insert the node under
346      * @param domcn
347      *          the child number to insert the node as
348      * @param charpos
349      *          the character position at which to insert the node
350      * @param ins
351      *          the node to be inserted
352      * @param doc
353      *          the document we are inserting into
354      */
355     _insertNode : function(siblings, parent, domcn, charpos, ins, doc) {
356       // siblings(domcn) is the node currently at the position we want to put
357       // the node
358 
359       if (domcn > siblings.length) {
360         throw new Error("Child number past end of nodes");
361       }
362       if (parent.nodeType != Node.ELEMENT_NODE && parent.nodeType != Node.DOCUMENT_NODE) {
363         throw new Error("Parent must be an element");
364       }
365 
366       if ((siblings.length > 0)) {
367 
368         // Check if inserting into text
369         if (this._prevNodeIsATextNode(siblings, domcn)) {
370           this._insertAtCharPos(charpos, siblings, domcn, ins, parent, doc);
371         } else if (domcn < siblings.length) {
372           parent.insertBefore(ins, siblings.item(domcn));
373         } else {
374           parent.appendChild(ins);
375         }
376       } else {
377         parent.appendChild(ins);
378       }
379     },
380 
381     /**
382      * Delete the appropriate amount of text from a Node.
383      * 
384      * Assumes a normalized document, i.e. no adjacent or empty text nodes.
385      * 
386      * @param delNode the text node to delete text from
387      * @param charpos the character position at which to delete
388      * @param length the number of characters to delete
389      * @param doc the document being deleted from
390      * @return the deleted text
391      */
392     _deleteText : function(delNode, charpos, length, doc) {
393       if (delNode.nodeType != Node.TEXT_NODE) {
394         throw new Error("Attempt to delete text from non-text node.");
395       }
396 
397       var text = delNode.nodeValue;
398       if (charpos > text.length) {
399         throw new Error("charpos past end of text node.");
400       }
401 
402       if ((length + charpos - 1) > text.length) {
403         throw new Error("length past end of text node.");
404       }
405 
406       var newText = text.substring(0, charpos - 1) + text.substring(charpos - 1 + length);
407       if (newText.length > 0) {
408         var newTextNode = doc.createTextNode(newText);
409         delNode.parentNode.insertBefore(newTextNode, delNode);
410       }
411 
412       delNode.parentNode.removeChild(delNode);
413 
414       return text.substring(charpos - 1, charpos - 1 + length);
415     },
416 
417     /**
418      * Delete the appropriate amount of text from a Node.
419      * 
420      * @param delNode the text node to delete text from
421      * @param charpos the character position at which to delete
422      * @param doc the document being deleted from
423      * @return the deleted text
424      */
425     _deleteText2 : function(delNode, charpos, doc) {
426       var length = delNode.nodeValue.length - charpos + 1;
427       return this._deleteText(delNode, charpos, length, doc);
428     }
429   }
