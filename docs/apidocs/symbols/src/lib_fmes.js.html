 11 /**
 12  * @class 
 13  * Fmes finds the differences between two DOM documents.
 14  *
 15  * Uses the Fast Match Edit Script algorithm (fmes).
 16  */
 17 Fmes = DiffXmlUtils.createClass(null, { 
 18   /**
 19    * Constructor
 20    * @constructs
 21    */
 22   init: function () {
 23   },
 24   proto : /** @lends Fmes.prototype */ {
 25     /**
 26      * Differences two DOM documents and returns the delta.
 27      *
 28      * @param document1    The original document
 29      * @param document2    The new document
 30      * @param delta        (optional) Delta object to be used
 31      *
 32      * @return A delta of changes
 33      */
 34     diff: function (document1, document2, delta) {
 35       var matchings = Match.easyMatch(document1, document2);
 36       return (new EditScript(document1, document2, matchings)).create(delta);
 37     } 
 38   }
 39 });
