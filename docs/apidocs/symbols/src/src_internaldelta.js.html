 11 /**
 12  * @class Default delta.
 13  * @extends Delta
 14  */
 15 InternalDelta = DiffXmlUtils.createClass(Delta, {
 16   /**
 17    * Constructor
 18    * @constructs
 19    */
 20   init: function (operations) {
 21     this._changes = operations||new Array();
 22   },
 23   proto : /** @lends InternalDelta.prototype */ {
 24     
 25     /**
 26      * Returns changes array
 27      * 
 28      * @returns changes array
 29      */
 30     getChanges: function () {
 31       return this._changes;
 32     },
 33 
 34     /**
 35      * Returns move operations as array
 36      * 
 37      * @returns move operations as array
 38      */
 39     getMoved: function () {
 40       return this._getChangesByType("move");
 41     },
 42     
 43     /**
 44      * Returns delete operations as array
 45      * 
 46      * @returns delete operations as array
 47      */
 48     getDeleted: function () {
 49       return this._getChangesByType("delete");
 50     },
 51     
 52     /**
 53      * Returns insert operations as array
 54      * 
 55      * @returns insert operations as array
 56      */
 57     getInserted: function () {
 58       return this._getChangesByType("insert");
 59     },
 60     
 61     /**
 62      * Returns update operations as array
 63      * 
 64      * @returns update operations as array
 65      */
 66     getUpdated: function () {
 67       return this._getChangesByType("update");
 68     },
 69 
 70     /**
 71      * Adds inserts for attributes of a node to an delta.
 72      * 
 73      * @param attrs the attributes to be added
 74      * @param path the path to the node they are to be added to
 75      */
 76     addAttrsToDelta: function (attrs, path) {
 77       var numAttrs;
 78       if (attrs == null) {
 79         numAttrs = 0;
 80       } else {
 81         numAttrs = attrs.length;
 82       }
 83 
 84       for (var i = 0; i < numAttrs; i++) {
 85         this.insert(attrs.item(i), path, 0, 1);
 86       }
 87     },
 88     
 89     /**
 90      * Appends an insert operation to the delta.
 91      * 
 92      * Set charpos to 1 if not needed.
 93      * 
 94      * @param n The node to insert
 95      * @param parent The path to the node to be parent of n
 96      * @param childno The child number of the parent node that n will become
 97      * @param charpos The character position to insert at
 98      */
 99     insert: function (n, parent, childno, charpos) {
100       if (parent instanceof Node) {
101         parent = NodeOps.getXPath(parent);
102       }
103 
104       var inserted = {
105         type: 'insert',
106         parent: parent,
107         nodeType: n.nodeType,
108         value: n.nodeValue
109       };
110       
111       if (n.nodeType != Node.ATTRIBUTE_NODE) {
112         inserted['childNo'] = childno;
113       } 
114       
115       if (n.nodeType == Node.ATTRIBUTE_NODE || n.nodeType == Node.ELEMENT_NODE || n.nodeType == Node.PROCESSING_INSTRUCTION_NODE) {
116         inserted['nodeName'] = n.nodeName;
117       }
118       
119       if (charpos > 1) {
120         inserted['charpos'] = charpos;
121       }
122       
123       if (n.nodeType == Node.ELEMENT_NODE) {
124         this.addAttrsToDelta(n.attributes, parent + "/node()[" + childno + "]");
125       }
126       
127       this._changes.push(inserted);
128     },
129     
130     /**
131      * Adds a delete operation to the delta for the given Node.
132      * 
133      * @param n The Node that is to be deleted
134      */
135     deleteNode: function(n) {
136       var deleted = {
137         type: 'delete',
138         node: NodeOps.getXPath(n)
139       };
140       
141       if (n.nodeType == Node.TEXT_NODE) {
142         var cn = new ChildNumber(n);
143         var charpos = cn.getXPathCharPos();
144         
145         if (charpos >= 1) {
146           deleted['charpos'] = charpos;
147           deleted['length'] = n.length;
148         }
149       }
150 
151       this._changes.push(deleted);
152     },
153 
154     /**
155      * Adds a Move operation to the delta. 
156      * 
157      * @param n The node being moved
158      * @param parent XPath to the new parent Node
159      * @param childno Child number of the parent n will become
160      * @param ncharpos The new character position for the Node
161      */
162     move: function (n, parent, childno, ncharpos) {
163       if (ncharpos < 1) {
164         throw new Error("New Character position must be >= 1");
165       }
166       
167       if (parent instanceof Node) {
168         parent = NodeOps.getXPath(parent);
169       }
170         
171       var moved = {
172         type: 'move',
173         node: NodeOps.getXPath(n),
174         ocharpos: new ChildNumber(n).getXPathCharPos(),
175         ncharpos: ncharpos,
176         parent: parent,
177         childNo: childno
178       };
179       
180       if (n.nodeType == Node.TEXT_NODE) {
181         moved['length'] = n.length;
182       }
183       
184       this._changes.push(moved);
185     },
186 
187     /**
188      * Adds an update operation to the delta.
189      * 
190      * @param w The node to update
191      * @param x The node to update it to
192      */
193     update: function (w, x) {
194       var updated = {
195         type: 'update',
196         node: NodeOps.getXPath(w)
197       };
198       
199       if (w.nodeType == Node.ELEMENT_NODE) {
200         updated['nodeName'] = x.nodeName;
201         this._updateAttributes(w, x);
202       } else {
203         updated['nodeValue'] = x.nodeValue;
204       }
205       
206       this._changes.push(updated);
207     },
208     
209     /**
210      * Updates the attributes of element w to be the same as x's.
211      * 
212      * @param w The Element to update the attributes of
213      * @param x The element holding the correct attributes
214      */
215     _updateAttributes: function (w, x) {
216       var wAttrs = w.attributes;
217       var xAttrs = x.attributes;
218       
219       //Delete any attrs of w not in x, update others
220       for (var i = 0; i < wAttrs.length; i++) {
221         var wAttr = wAttrs.item(i);
222         var xAttr = xAttrs[wAttr.name];
223         if (xAttr == null) {
224           this.deleteNode(wAttrs.item(i));
225         } else if (wAttr.nodeValue != xAttr.nodeValue) {
226           this.update(wAttr, xAttr);
227         }
228       }
229         
230       //Add any attrs in x but not w
231       for (var j = 0; j < xAttrs.length; j++) {
232         var xAttr = xAttrs.item(j);
233         if (wAttrs[xAttr.name] == null) {
234           this.insert(xAttr, NodeOps.getXPath(w), 0, 1);
235         }
236       }
237     },
238     
239     _getChangesByType: function (type) {
240       var result = new Array();
241       for (var i = 0, l = this._changes.length; i < l; i++) {
242         if (this._changes[i].type == type) {
243           result.push(this._changes[i]);
244         }
245       }
246       return result;
247     }
248   }
249 });
