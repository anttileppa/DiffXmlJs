 11 /**
 12  * @class
 13  * Solves the "good matchings" problem for the FMES algorithm.
 14  *
 15  * Essentially pairs nodes that match between documents.
 16  * Uses the "fast match" algorithm is detailed in the paper
 17  * "Change Detection in Hierarchically Structure Information".
 18  *
 19  * WARNING: Will only work correctly with acylic documents.
 20  * TODO: Add alternate matching code for cylic documents.
 21  * See: http://www.devarticles.com/c/a/Development-Cycles/How-to-Strike-a-Match/
 22  * for information on how to match strings.
 23  */
 24 Match = /** @lends Match */ {
 25   /**
 26    * Performs fast match algorithm on given DOM documents.
 27    * 
 28    *  TODO: May want to consider starting at same point in 2nd tree somehow, 
 29    *  may lead to better matches.
 30    * 
 31    * @param document1 The original document
 32    * @param document2 The modified document
 33    * 
 34    * @return NodeSet containing pairs of matching nodes.
 35    */
 36   easyMatch: function (document1, document2) {
 37     var matchSet = new NodePairs();
 38     
 39     document1.documentElement.normalize();
 40     document2.documentElement.normalize();
 41     
 42     var list1 = this._initialiseAndOrderNodes(document1);
 43     var list2 = this._initialiseAndOrderNodes(document2);
 44     
 45     //Explicitly add document elements and root
 46     matchSet.add(document1, document2);
 47     matchSet.add(document1.documentElement, document2.documentElement);
 48     
 49     // Proceed bottom up on List 1
 50     for (var i = 0, l = list1.length; i < l; i++) {
 51       var nd1 = list1[i];
 52       var n1 = nd1.getNode();
 53       for (var j = 0, jl = list2.length; j < jl; j++) {
 54         var nd2 = list2[j];
 55         var n2 = nd2.getNode();
 56         
 57         if (this._compareNodes(n1, n2)) {
 58           matchSet.add(n1, n2);
 59           //Don't want to consider it again
 60           this._removeItemFromNodeDepth(list2, nd2);
 61           break;
 62         }
 63       }
 64     }
 65 
 66     return matchSet;
 67   },
 68   
 69   /**
 70    * Compares two elements two determine whether they should be matched.
 71    * 
 72    * TODO: This method is critical in getting good results. Will need to be
 73    * tweaked. In addition, it may be an idea to allow certain doc types to
 74    * override it. Consider comparing position, matching of kids etc.
 75    * 
 76    * @param a First element
 77    * @param b Potential match for b
 78    * @return true if nodes match, false otherwise
 79    */
 80   compareElements: function (a, b) {
 81     var ret = false;
 82 
 83     if (a.nodeName == b.nodeName) {
 84         //Compare attributes
 85         
 86         //Attributes are equal until we find one that doesn't match
 87         ret = true;
 88         
 89         var aAttrs = a.attributes;
 90         var bAttrs = b.attributes;
 91 
 92         var numberAAttrs = 0;
 93         if (aAttrs != null) {
 94             numberAAttrs = aAttrs.length;
 95         }
 96         var numberBAttrs = 0;
 97         if (bAttrs != null) {
 98             numberBAttrs = bAttrs.length;
 99         }
100         if (numberAAttrs != numberBAttrs) {
101             ret = false;
102         }
103 
104         var i = 0;
105         while (ret && (i < numberAAttrs)) {
106             // Check if attr exists in other tag
107             var bItem = bAttrs[aAttrs.item(i).nodeName]; 
108             if (bItem == null || (bItem.value != aAttrs.item(i).value)) {
109                 ret = false;
110             } 
111             i++;
112         }
113     }
114     
115     return ret;
116   },
117   
118   /**
119    * Compares two text nodes to determine if they should be matched.
120    * 
121    * Takes into account whitespace options.
122    * 
123    * @param a First node
124    * @param b Potential match for a
125    * @return True if nodes match, false otherwise
126    */
127 
128   compareTextNodes: function (a, b) {
129     var aString = a.data;
130     var bString = b.data;
131 
132     return aString == bString;
133   },
134 
135   /**
136    * Compares 2 nodes to determine whether they should match.
137    * 
138    * TODO: Check if more comparisons are needed
139    * TODO: Consider moving out to a separate class, implementing an interface
140    * 
141    * @param a first node
142    * @param b potential match for a
143    * @return true if nodes match, false otherwise
144    */
145   _compareNodes: function (a, b) {
146     var ret = false;
147 
148     if (a.nodeType == b.nodeType) { 
149       switch (a.nodeType) {
150         case Node.ELEMENT_NODE :
151           ret = this.compareElements(a, b);
152         break;
153         case Node.TEXT_NODE :
154           ret = this.compareTextNodes(a, b);
155         break;
156         case Node.DOCUMENT_NODE :
157           //Always match document nodes
158           ret = true;
159         break;
160         default :
161           ret = a.nodeValue == b.nodeValue;
162         break;
163       }
164     }
165     
166     return ret;
167   },
168 
169   /**
170    * Returns a list of Nodes sorted according to their depths.
171    * 
172    * Does *NOT* include root or documentElement
173    * 
174    * TreeSet is sorted in reverse order of depth according to
175    * NodeInfoComparator.
176    * 
177    * @param doc The document to be initialised and ordered.
178    * @return A depth-ordered list of the nodes in the doc.
179    */
180   _initialiseAndOrderNodes: function (doc) {
181     var ni = doc.createNodeIterator(doc, NodeFilter.SHOW_ALL, null);
182     var depthSorted = new Array();
183     var n;
184     while ((n = ni.nextNode()) != null) {
185       if (!(NodeOps.checkIfSameNode(doc, n) || NodeOps.checkIfSameNode(doc.documentElement, n) || n.nodeType == Node.DOCUMENT_TYPE_NODE)) {
186         depthSorted.push(new NodeDepth(n));
187       }
188     }
189   
190     ni.detach();
191   
192     depthSorted.sort(function (nodeInfo1, nodeInfo2) {
193       return nodeInfo2.getDepth() - nodeInfo1.getDepth();
194     });
195   
196     return depthSorted;
197   },
198   
199   /**
200    * Removes nodeDepth from array of nodeDepths
201    * 
202    * @param list array of nodeDepths
203    * @param nodeDepth nodeDepth to be removed
204    */
205   _removeItemFromNodeDepth: function (list, nodeDepth) {
206     var index = list.indexOf(nodeDepth);
207     if (index != -1) {
208       list.splice(index, 1);
209     }
210   }
211 
212 };
