 11 /**
 12  * @class Creates the edit script for the fmes algorithm.
 13  *
 14  * Uses the algorithm described in the paper
 15  * "Change Detection in Hierarchically Structure Information".
 16  */
 17 EditScript = DiffXmlUtils.createClass(null, {  
 18   
 19   /**
 20    * Constructor for EditScript.
 21    * Used to create a list of modifications that will turn document1 into document2,
 22    * given a set of matching nodes.
 23    * 
 24    * @param document1 the original document
 25    * @param document2 the modified document
 26    * @param matchings the set of matching nodes
 27    */
 28   init: function (document1, document2, matchings) {
 29     /**
 30      * The original document.
 31      */
 32     this._document1 = document1;
 33     /**
 34      * The modified document.
 35      */
 36     this._document2 = document2;
 37     /**
 38      * The set of matching nodes.
 39      */
 40     this._matchings = matchings;
 41     /**
 42      * Delta
 43      */
 44     this._delta = null;
 45   },
 46   proto : /** @lends EditScript.prototype */ {
 47     
 48     /**
 49      * Creates an Edit Script conforming to matchings that transforms
 50      * document1 into document2.
 51      *
 52      * Uses algorithm in "Change Detection in Hierarchically Structured
 53      * Information".
 54      *
 55      * @return the resultant Edit Script
 56      */
 57     create: function(delta) {
 58       this._delta = delta||new DefaultDelta();
 59       
 60       // Fifo used to do a breadth first traversal of doc2
 61       var fifo = new NodeFifo();
 62       fifo.addChildrenOfNode(this._document2);
 63       
 64       var doc2docEl = this._document2.documentElement;
 65       //Special case for aligning children of root node
 66       this._alignChildren(this._document1, this._document2, this._matchings);
 67 
 68       while (!fifo.isEmpty()) {
 69           var x = fifo.pop();
 70           fifo.addChildrenOfNode(x);
 71 
 72           var y = x.parentNode;
 73           var z = this._matchings.getPartner(y);
 74           var w = this._matchings.getPartner(x);
 75 
 76           if (!this._matchings.isMatched(x)) {
 77             w = this._doInsert(x, z);
 78           } else {
 79             // TODO: Update should go here
 80             // Special case for document element
 81               if (NodeOps.checkIfSameNode(x, doc2docEl) && !Match.compareElements(w, x)) {
 82                 w = this._doUpdate(w, x);
 83               } else { 
 84                 if (!(this._matchings.getPartner(y) == w.parentNode)) {
 85                   this._doMove(w, x, z, this._matchings);
 86                 }
 87               }
 88           }
 89 
 90           this._alignChildren(w, x, this._matchings);
 91       }
 92 
 93       this._deletePhase(this._document1, this._matchings);
 94 
 95       // TODO: Assert following
 96       // Post-Condition es is a minimum cost edit script,
 97       // Matchings is a total matching and
 98       // doc1 is isomorphic to doc2
 99 
100       return this._delta;
101     },
102 
103     /**
104      * Updates a Node to the value of another node.
105      * 
106      * @param w The Node to be updated
107      * @param x The Node to make it like
108      * @return The new Node
109      */
110     _doUpdate: function (w, x) {
111       var doc1 = w.ownerDocument;
112       var newW = null;
113       if (w.nodeType == Node.ELEMENT_NODE) {
114 
115           this._delta.update(w, x);
116 
117           //Unfortunately, you can't change the node name in DOM, so we need
118           //to create a new node and copy it all over
119           
120           //TODO: Note you don't actually *need* to do this!!!
121           //TODO: Only call this when in debug
122           newW = doc1.createElement(x.nodeName);
123           
124           // Copy x's attributes to the new element
125           var attrs = x.attributes;
126           for (var i = 0; i < attrs.length; i++) {
127             newW.attributes[attrs[i].name] = attrs[i].value;
128           }
129           
130           while (w.hasChildNodes()) {
131             newW.appendChild(w.firstChild);
132           }
133 
134           w.parentNode.replaceChild(newW, w);
135           this._matchings.remove(w);
136           this._matchings.add(newW, x);   
137       }
138       
139       return newW;
140     },
141     /**
142      * Inserts (clone of) node x as child of z according to the algorithm 
143      * and updates the Edit Script.
144      *
145      * @param x          current node
146      * @param z          partner of x's parent
147      * @return           the inserted node
148      */
149     _doInsert: function (x, z) {
150         //Find the child number (k) to insert w as child of z 
151         var pos = new FindPosition(x, this._matchings);
152 
153         //Apply insert to doc1
154         //The node we want to insert is the import of x with attributes but no children
155 
156         var w = x.cloneNode(false);
157         
158         //Need to set in order as won't be revisited
159         NodeOps.setInOrder(w);
160         NodeOps.setInOrder(x);
161 
162         this._delta.insert(w, z, pos.getXPathInsertPosition(), pos.getCharInsertPosition());
163 
164         //Take match of parent (z), and insert
165         w = this._insertAsChild(pos.getDOMInsertPosition(), z, w);
166 
167         this._matchings.add(w, x);
168 
169         return w;
170     },
171 
172     /**
173      * Performs a move operation according to the algorithm and updates
174      * the EditScript.
175      *
176      * @param w          the node to be moved
177      * @param x          the matching node
178      * @param z          the partner of x's parent
179      * @param matchings  the set of matching nodes
180      */
181     _doMove: function (w, x, z, matchings) {
182       
183       var v = w.parentNode;
184       var y = x.parentNode;
185       
186       // Apply move if parents not matched and not null
187 
188       var partnerY = matchings.getPartner(y);
189       if (NodeOps.checkIfSameNode(v, partnerY)) {
190         throw new Error("v is same as partnerY");
191       }
192       
193       var pos = new FindPosition(x, matchings);
194       
195       NodeOps.setInOrder(w);
196       NodeOps.setInOrder(x);
197 
198       this._delta.move(w, z, pos.getXPathInsertPosition(), pos.getCharInsertPosition());
199 
200       //Apply move to T1
201       this._insertAsChild(pos.getDOMInsertPosition(), z, w);
202     },
203 
204     /**
205      * Performs the deletePhase of the algorithm.
206      *
207      * @param n          the current node
208      * @param matchings  the set of matching nodes
209      */
210     _deletePhase: function (n, matchings) {
211       var kids = n.childNodes;
212       if (kids != null) {
213         // Note that we loop *backward* through kids
214         for (var i = (kids.length - 1); i >= 0; i--) {
215           this._deletePhase(kids.item(i), matchings);
216         }
217       }
218 
219       // If node isn't matched, delete it
220       if (!matchings.isMatched(n) && n.nodeType != Node.DOCUMENT_TYPE_NODE) {
221         this._delta.deleteNode(n);
222         n.parentNode.removeChild(n);
223       }
224     },
225 
226     /**
227      * Mark the children of a node out of order.
228      *
229      * @param n the parent of the nodes to mark out of order
230      */
231     _markChildrenOutOfOrder: function (n) {
232       for (var i = 0, l = n.childNodes.length; i < l; i++) {
233         NodeOps.setOutOfOrder(n.childNodes.item(i));
234       }
235     },
236 
237     /**
238      * Mark the children of a node in order.
239      *
240      * @param n the parent of the nodes to mark in order
241      */
242     _markChildrenInOrder: function (n) {
243       for (var i = 0, l = n.childNodes.length; i < l; i++) {
244         NodeOps.setInOrder(n.childNodes.item(i));
245       }
246     },
247     
248     /**
249      * Marks the Nodes in the given list and their partners "inorder".
250      *
251      * @param seq  the Nodes to mark "inorder"
252      * @param matchings the set of matching Nodes
253      */
254     _setNodesInOrder: function (seq, matchings) {
255       for (var i = 0, l = seq.length; i < l; i++) {
256         var node = seq[i];
257         NodeOps.setInOrder(node);
258         NodeOps.setInOrder(matchings.getPartner(node));
259       }
260     },
261 
262     /**
263      * Moves nodes that are not in order to correct position.
264      *
265      * @param w Node with potentially misaligned children
266      * @param wSeq Sequence of children of w that have matches in the children of x
267      * @param stay The List of nodes not to be moved
268      * @param matchings The set of matching nodes
269      */
270     _moveMisalignedNodes: function (w, wSeq, stay, matchings) {
271       for (var wSeqIndex = 0, wSeqLength = wSeq.length; wSeqIndex < wSeqLength; wSeqIndex++) {
272         var a = wSeq[wSeqIndex];
273         if (NodeOps.getNodeIndex(stay, a) == -1) {
274           var b = matchings.getPartner(a);
275           var pos = new FindPosition(b, matchings);
276           this._delta.move(a, w, pos.getXPathInsertPosition(), pos.getCharInsertPosition());
277           this._insertAsChild(pos.getDOMInsertPosition(), w, a);
278           NodeOps.setInOrder(a);
279           NodeOps.setInOrder(b);
280         }
281       }
282     },
283 
284     /**
285      * Aligns children of current node that are not in order.
286      *
287      * @param w  the match of the current node.
288      * @param x  the current node
289 
290      * @param matchings  the set of matchings
291      */
292     _alignChildren: function (w, x, matchings) {
293       //Order of w and x is important
294       this._markChildrenOutOfOrder(w);
295       this._markChildrenOutOfOrder(x);
296 
297       var wKids = w.childNodes;
298       var xKids = x.childNodes;
299 
300       var wSeq = NodeSequence.getSequence(wKids, xKids, matchings);
301       var xSeq = NodeSequence.getSequence(xKids, wKids, matchings);
302 
303       var lcsSeq = NodeSequence.getLCS(wSeq, xSeq, matchings);
304       this._setNodesInOrder(lcsSeq, matchings);
305       
306       this._moveMisalignedNodes(w, wSeq, lcsSeq, matchings);
307       
308       //The following is missing from the algorithm, but is important
309       this._markChildrenInOrder(w);
310       this._markChildrenInOrder(x);
311     },
312     
313     /**
314      * Inserts a given node as numbered child of a parent node.
315      *
316      * If childNum doesn't exist the node is simply appended.
317      *
318      * @param childNum  the position to add the node to
319      * @param parent    the node that is to be the parent
320      * @param insNode   the node to be inserted
321      * @return The inserted Node
322      */
323     _insertAsChild: function (childNum, parent, insNode) {
324       if (insNode.parentNode) {
325         insNode.parentNode.removeChild(insNode);
326       }
327       
328       var child = parent.childNodes.item(childNum);
329       if (child) {
330         parent.insertBefore(insNode, child);
331       } else {
332         parent.appendChild(insNode);
333       }
334       
335       return insNode;
336     }
337   }
