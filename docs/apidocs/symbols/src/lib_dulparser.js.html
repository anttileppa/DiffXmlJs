 11 /**
 12  * @class
 13  * Parser for DUL Patch format
 14  */
 15 DULParser = /** @lends DULParser */ {
 16   
 17   /**
 18    * Parses DUL format patch and converts it into internal format
 19    * 
 20    * @param dulPatch DUL patch
 21    * @returns InternalDelta
 22    */
 23   parseDULPatch: function (dulPatch) {
 24     var deltaElement = dulPatch.documentElement;
 25     if (deltaElement.nodeName != DULConstants.DELTA) {
 26       throw new Error("All deltas must begin with a " + DULConstants.DELTA + " element.");
 27     }
 28 
 29     var processed = new Array();
 30     
 31     // TODO: 
 32     // var siblingContext = deltaElement.attributes[DULConstants.SIBLING_CONTEXT];
 33     // var parentContext = deltaElement.attributes[DULConstants.PARENT_CONTEXT];
 34     // var parentSiblingContext = deltaElement.attributes[DULConstants.PARENT_SIBLING_CONTEXT];
 35     // var parentSiblingContext = deltaElement.attributes[DULConstants.REVERSE_PATCH];
 36     // var resolveEntities = deltaElement.attributes[DULConstants.RESOLVE_ENTITIES];
 37     
 38     for (var i = 0, l = deltaElement.childNodes.length; i < l; i++) {
 39       var operationNode = deltaElement.childNodes[i];
 40       switch (operationNode.nodeName) {
 41         case DULConstants.INSERT:
 42           processed.push(this._parseInsert(operationNode));
 43         break;
 44         case DULConstants.DELETE:
 45           processed.push(this._parseDelete(operationNode));
 46         break;
 47         case DULConstants.MOVE:
 48           processed.push(this._parseMove(operationNode));
 49         break;
 50         case DULConstants.UPDATE:
 51           processed.push(this._parseUpdate(operationNode));
 52         break;
 53       }
 54     }
 55     
 56     var delta = new InternalDelta(processed);
 57     return delta;
 58   },
 59   
 60   _parseInsert: function (operation) {
 61     var parent = operation.getAttribute(DULConstants.PARENT);
 62     var nodeType = parseInt(operation.getAttribute(DULConstants.NODETYPE));
 63 
 64     var internalOperation = {
 65       type: 'insert',
 66       parent: parent,
 67       nodeType: nodeType
 68     };
 69     
 70     if (nodeType != Node.ATTRIBUTE_NODE) {
 71       internalOperation['childNo'] = operation.getAttribute(DULConstants.CHILDNO);
 72     }
 73     
 74     if ((nodeType == Node.ATTRIBUTE_NODE) || (nodeType == Node.ELEMENT_NODE) || (nodeType == Node.PROCESSING_INSTRUCTION_NODE)) {
 75       internalOperation['nodeName'] = operation.getAttribute(DULConstants.NAME);
 76     }
 77     
 78     var charPosAttr = operation.attributes.getNamedItem(DULConstants.CHARPOS);
 79     if (charPosAttr) {
 80       internalOperation['charpos'] = charPosAttr.value;
 81     }
 82     
 83     var valueNode = operation.firstChild;
 84     if (valueNode) {
 85       internalOperation['value'] = valueNode.nodeValue;
 86     };
 87 
 88     return internalOperation;
 89   },
 90   
 91   _parseDelete: function (operation) {
 92     var internalOperation = {
 93       type: 'delete',
 94       node: operation.getAttribute(DULConstants.NODE)
 95     };
 96     
 97     var charpos = operation.hasAttribute(DULConstants.CHARPOS) ? parseInt(operation.getAttribute(DULConstants.CHARPOS)) : NaN;
 98     var length = operation.hasAttribute(DULConstants.LENGTH) ? parseInt(operation.getAttribute(DULConstants.LENGTH)) : NaN;
 99 
100     if (!isNaN(charpos)) {
101       internalOperation['charpos'] = parseInt(charpos);
102     }
103     
104     if (!isNaN(length)) {
105       internalOperation['length'] = parseInt(length);
106     }
107     
108     return internalOperation;
109   },
110   
111   _parseMove: function (operation) {
112     var internalOperation = {
113       type: 'move',
114       node: operation.getAttribute(DULConstants.NODE),
115       ocharpos: parseInt(operation.getAttribute(DULConstants.OLD_CHARPOS)),
116       ncharpos: parseInt(operation.getAttribute(DULConstants.NEW_CHARPOS)),
117       parent: operation.getAttribute(DULConstants.PARENT),
118       childNo: parseInt(operation.getAttribute(DULConstants.CHILDNO))
119     };
120     
121     var lengthAttribute = operation.getAttribute(DULConstants.LENGTH);
122     if (lengthAttribute !== null) {
123       internalOperation['length'] = parseInt(lengthAttribute);
124     };
125     
126     return internalOperation;
127   },
128   
129   _parseUpdate: function (operation) {
130     var internalOperation = {
131       type: 'update',
132       node: operation.getAttribute(DULConstants.NODE)
133     };
134     
135     // We do not know is the node element or something else so we add operation node content to 
136     // both nodeName and nodeValue fields
137     // this should be changed in future.
138     
139     internalOperation['nodeName'] = internalOperation['nodeValue'] = operation.firstChild.nodeValue;
140       
141     return internalOperation;
142   }
143   
144 };
