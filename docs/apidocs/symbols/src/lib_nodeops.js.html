1 /**
  2  * @class 
  3  * Class to handle general diffxml operations on Nodes.
  4  */
  5 NodeOps = /** @lends NodeOps */ {
  6     
  7   /**
  8    * Replacement for obsolete Node.setUserData
  9    * 
 10    * @param node node
 11    * @param name userKey
 12    * @param value userData
 13    */
 14   setUserData: function (node, name, value) {
 15     // TODO: Use Element.dataSet (https://developer.mozilla.org/en-US/docs/DOM/element.dataset) when supported
 16     
 17     if (!node._userData) {
 18       node._userData = {};
 19     }
 20     
 21     node._userData[name] = value;
 22   },
 23   
 24   /**
 25    * Replacement for obsolete Node.getUserData
 26    * 
 27    * @param node node
 28    * @param name userKey
 29    */
 30   getUserData: function (node, name) {
 31     // TODO: Use Element.dataSet (https://developer.mozilla.org/en-US/docs/DOM/element.dataset) when supported
 32 
 33     if (node._userData) {
 34       return node._userData[name];
 35     }
 36     
 37     return null;
 38   },
 39   
 40   /**
 41    * Returns a index of a node from array of nodes or NodeList
 42    * 
 43    * @param list array / NodeList
 44    * @param node node
 45    * @returns index of node or -1 if not found
 46    */
 47   getNodeIndex: function (list, node) {
 48     // TODO: Is this method really necessary?
 49     if (list instanceof Array) {
 50       return list.indexOf(node);
 51     } else if (list instanceof NodeList) {
 52       for (var i = 0, l = list.length; i < l; i++) {
 53         if (list.item(i) == node) {
 54           return i;
 55         }
 56       }
 57     }
 58     
 59     return -1;
 60   },
 61 
 62   /**
 63    * Mark the node as being "inorder".
 64    *
 65    * @param n the node to mark as "inorder"
 66    */
 67   setInOrder: function (n) {
 68     NodeOps.setUserData(n, "inorder", true);
 69   },
 70 
 71   /**
 72    * Mark the node as not being "inorder".
 73    *
 74    * @param n the node to mark as not "inorder"
 75    */
 76   setOutOfOrder: function (n) {
 77     NodeOps.setUserData(n, "inorder", false);
 78   },
 79 
 80   /**
 81    * Check if node is marked "inorder".
 82    *
 83    * Note that nodes are inorder by default.
 84    *
 85    * @param n node to check
 86    * @return false if UserData set to False, true otherwise
 87    */
 88   isInOrder: function (n) {
 89     var data = NodeOps.getUserData(n, "inorder");
 90     return data == null ? true : data;
 91   },
 92 
 93   /**
 94    * Check if nodes are the same.
 95    * 
 96    * @param x first node to check
 97    * @param y second node to check
 98    * @return true if same node, false otherwise
 99    */
100 
101   checkIfSameNode: function (x, y) {
102     if (x != null && y != null) {
103       return x == y;
104     } 
105 
106     return false;
107   },
108   
109   /**
110    * Calculates an XPath that uniquely identifies the given node.
111    * For text nodes note that the given node may only be part of the returned
112    * node due to coalescing issues; use an offset and length to identify it
113    * unambiguously.
114    * 
115    * @param n The node to calculate the XPath for.
116    * @return The XPath to the node as a String
117    */
118   getXPath: function(n) {
119     var xpath;
120     if (n.nodeType == Node.ATTRIBUTE_NODE) {
121         //Slightly special case for attributes as they are considered to
122         //have no parent
123       // TODO: ownerElement property is deprecated.
124       xpath = this.getXPath(n.ownerElement) + "/@" + n.nodeName;
125     } else if (n.nodeType == Node.DOCUMENT_NODE) {
126       xpath = "/";
127     } else if (n.nodeType == Node.DOCUMENT_TYPE_NODE) {
128       throw new Error("DocumentType nodes cannot be identified with XPath");
129     } else if (n.parentNode.nodeType == Node.DOCUMENT_NODE) {
130       var cn = new ChildNumber(n);
131       xpath = "/node()[" + cn.getXPath() + "]"; 
132     } else {
133       var cn = new ChildNumber(n);
134       xpath = this.getXPath(n.parentNode) + "/node()[" + cn.getXPath() + "]";
135     }
136     
137     return xpath;
138   },
139   
140   /**
141    * Check if node is an empty text node.
142    * 
143    * @param n The Node to test.
144    * @return True if it is a 0 sized text node
145    */
146   nodeIsEmptyText: function (n) {
147     if (n.nodeType === Node.TEXT_NODE) {
148       return n.length === 0;
149     }
150     
151     return false;
152   }
153 };
