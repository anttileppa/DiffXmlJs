 11 /**
 12  * @class 
 13  * Finds the position to insert a Node at.
 14  * 
 15  * Calculates XPath, DOM and character position.
 16  */
 17 FindPosition = DiffXmlUtils.createClass(null, {
 18   
 19   /**
 20    * Finds the child number to insert a node as.
 21    *
 22    * (Equivalent to the current child number of the node to insert
 23    * before)
 24    *
 25    * @param x         the node with no partner
 26    * @param matchings the set of matching nodes
 27    */
 28   init: function (x, matchings) {
 29     
 30     /** The DOM position. */
 31     this._insertPositionDOM = null;
 32     
 33     /** The XPath position. */
 34     this._insertPositionXPath = null;
 35     
 36     /** The character position. */
 37     this._charInsertPosition = null;
 38     
 39     var v = this._getInOrderLeftSibling(x);
 40     if (v == null) {
 41       this._insertPositionDOM = 0;
 42       this._insertPositionXPath = 1;
 43       this._charInsertPosition = 1;
 44     } else {
 45       /**
 46         * Get partner of v and return index after
 47         * (we want to insert after the previous in-order node, so that
 48         * w's position is equivalent to x's).
 49         */
 50       var u = matchings.getPartner(v);
 51       var uChildNo = new ChildNumber(u);
 52       //Need position after u
 53       this._insertPositionDOM = uChildNo.getInOrderDOM() + 1;
 54       this._insertPositionXPath = uChildNo.getInOrderXPath() + 1;
 55       //For xpath, character position is used if node is text node
 56       if (u.nodeType == Node.TEXT_NODE) {
 57         this._charInsertPosition = uChildNo.getInOrderXPathCharPos() + u.length;
 58       } else {
 59        this._charInsertPosition = 1;
 60       }
 61     }
 62   },
 63   proto : /** @lends FindPosition.prototype */ {
 64     /**
 65      * Gets the rightmost left sibling of n marked "inorder".
 66      *
 67      * @param n Node to find "in order" left sibling of
 68      * @return  Either the "in order" left sibling or null if none
 69      */
 70     _getInOrderLeftSibling: function (n) {
 71       var curr = n.previousSibling;
 72       while (curr != null && !NodeOps.isInOrder(curr)) {
 73         curr = curr.previousSibling;
 74       }
 75 
 76       return curr;
 77     },
 78 
 79     /**
 80      * Returns the DOM number the node should have when inserted.
 81      * 
 82      * @return the DOM number to insert the node as
 83      */
 84     getDOMInsertPosition: function () {
 85       return this._insertPositionDOM;
 86     },
 87     
 88     /**
 89      * Returns the XPath number the node should have when inserted.
 90      * 
 91      * @return The XPath number to insert the node as
 92      */
 93     getXPathInsertPosition: function () {
 94       return this._insertPositionXPath;
 95     },
 96     
 97     /**
 98      * Returns the character position to insert the node as.
 99      * 
100      * @return The character position to insert the node at
101      */
102     getCharInsertPosition: function () {
103       return this._charInsertPosition;
104     }
105   }  
106 });
